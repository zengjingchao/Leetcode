class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        if(nums.size()==0)
            return 0;
        int p1=0,p2=0;
       while(p2<nums.size()){
           while(nums[p2]==nums[p1]){
               p2++;
           }
           p1++;
           nums[p1]=nums[p2];
       }
        return p1;  //这里返回的是最后一个有效元素的下标，而不是有效元素个数
    }
};
