/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool hasCycle(ListNode *head) {
       set<ListNode*> sett;
        int cnt=0;
        if(head==NULL){
            return false;
        }
        while(head){
            sett.insert(head);
            cnt++;
            if(cnt!=sett.size()){
                return true;
            }
            head=head->next;
        }
        return false;
    }
};
