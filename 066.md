class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        int len=digits.size();
        int carry=1;
        for(int i=len-1;i>=0;i--){
            if(digits[i]>=0&&digits[i]<=8){
                digits[i]++;
                return digits;
            }else if(digits[i]>=9){
                digits[i]=(digits[i]+carry)%10;
                carry=1;
            }
        }
        if(carry==1){
            digits.insert(digits.begin(),1);
        }
        return digits;
    }
};
