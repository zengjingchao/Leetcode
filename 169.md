class Solution {
public:
    int majorityElement(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        int temp=1,maxValue=0,ans=nums[0];
        for(int i=1;i<nums.size();i++){
            if(nums[i]==nums[i-1]){
                temp++;
            }else{
                temp=1;
            }
            if(temp>maxValue){
                maxValue=temp;
                ans=nums[i];
            }
        }
        return ans;
    }
};
