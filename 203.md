/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeElements(ListNode* head, int val) {
        if(head==NULL)
            return head;
     ListNode* p=head;
        while(p){
            while(p!=NULL&&p->next!=NULL&&p->val==val){
                p->val=p->next->val;
                p->next=p->next->next;
            }
            if(p->val==val&&p->next==NULL){
                ListNode* pp=head;
                if(pp==p)
                    return NULL;
                while(pp->next!=p){
                    pp=pp->next;
                }
                pp->next=NULL;
            }
            p=p->next;
        }
        return head;
    }
};
