class Solution {
public:
    bool containsDuplicate(vector<int>& nums) {
        bool flag=false;
        set<int> s;
        for(int i=0;i<nums.size();i++){
            s.insert(nums[i]);
            if(i+1!=s.size()){
                flag=true;
                break;
            }
        }
        return flag;
    }
};
